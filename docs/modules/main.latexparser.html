example
import {latexParser as lp} from 'latex-utensils'

const ast = lp.parse('a $b+c$ d')
const pat = lp.pattern(lp.isInlienMath)

const ret0 = pat.matchAll(ast.content)
const ret1 = lp.find(ast.content, lp.isInlienMath)

    default

    example
